Useful Starter Queries

SELECT * FROM datadock_demo_people;
SELECT d.name AS department, COUNT(p.id) AS people
FROM datadock_demo_departments d
JOIN datadock_demo_people p ON p.department_id = d.id
GROUP BY d.name;
SELECT p.name, pr.name AS project, pr.status, pr.budget
FROM datadock_demo_people p
JOIN datadock_demo_projects pr ON pr.owner_id = p.id;
SELECT project_id, SUM(amount) AS event_total
FROM datadock_demo_events
GROUP BY project_id;
SELECT pr.name, SUM(i.amount) AS invoiced
FROM datadock_demo_projects pr
JOIN datadock_demo_invoices i ON i.project_id = pr.id
GROUP BY pr.name;
